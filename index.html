const natural = require('natural'); const tokenizer = new natural.WordTokenizer(); const text = "This product is very good and amazing"; const words = tokenizer.tokenize(text); console.log("Imported natural package successfully"); console.log("Original Text:", text); console.log("Tokenized Words:", words); const analyzer = new natural.SentimentAnalyzer( "English", natural.PorterStemmer, "afinn" ); const score = analyzer.getSentiment(words); console.log("Sentiment Score:", score);
